# Basic commands

#Search for public images on the Docker hub and return info
docker search [image name] 

example: docker search ubuntu
---
#List all local images
docker images 
---
#Create container with given image and specified commands
docker run [OPTIONS] [image name] [COMMAND] [ARG...] 

example: docker run -i -t ubuntu
--- 
#List running containers
docker ps 

#List all containers including stopped ones
docker ps --all 
---
#Stops container
docker stop [container id or container name] 
---
#Restarts container
docker restart [container id] 
---
docker rename [old_name] [new_name]
---
#Access our container and get a bash shell inside the container
docker exec -it [container_name] bash 
---
#Fetch logs from the specified container
docker logs [container_name] 
---
#Remove container
docker rm [container_name] 

Note: Either stop container first and then remove or use -f flag after the rm command
---
#Remove image
docker rmi [image_id]
